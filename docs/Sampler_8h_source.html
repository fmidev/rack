343 x = voidMarker;

349 values[minusStr+quantity] = -x;

351

353 //if (dataOk || (skipVoid==0)){

355 formatter.toStream(ostr, variableMap, -1); // leaves ${variable} ?

361 // formatter.expand(map, true);
