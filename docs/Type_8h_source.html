507 template <class C>
508 const std::type_info & Type::guessArrayType(const C & l){
509 
510 
511  typedef std::set<const std::type_info *> typeset;
512 
513  typeset s;
514  for (typename C::const_iterator it = l.begin(); it != l.end(); ++it) {
515  s.insert(& guessType(*it));
516  }
517 
519  if (s.find(& typeid(std::string)) != s.end())
520  return typeid(std::string);
521 
523  if (s.find(& typeid(double)) != s.end())
524  return typeid(double);
525 
526  if (s.find(& typeid(int)) != s.end())
527  return typeid(int);
528 
530  if (s.find(& typeid(bool)) != s.end())
531  return typeid(bool);
532 
533  // General fallback solution
534  return typeid(std::string);
535 
536 }
