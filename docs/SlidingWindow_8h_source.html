68 void setSlidingMode(bool horzMultiple, bool vertMultiple){
69
70 modeStr = "??";
71 modeStr.at(0) = horzMultiple?'H':'h';
72 modeStr.at(1) = vertMultiple?'V':'v';
73
74 updateHorz = horzMultiple ? &SlidingWindow<C,R>::updateHorzMultiple : &SlidingWindow<C,R>::updateHorzSingle;
75 updateVert = vertMultiple ? &SlidingWindow<C,R>::updateVertMultiple : &SlidingWindow<C,R>::updateVertSingle;
76
77 if (horzMultiple && vertMultiple)
78 fill = & SlidingWindow<C,R>::fillBoth;
79 else if (horzMultiple)
80 fill = & SlidingWindow<C,R>::fillVert; //Horz;
81 else if (vertMultiple)
82 fill = & SlidingWindow<C,R>::fillHorz; // Vert;
83 else {
84 drain::Logger mout(getImgLog(), __FILE__, __FUNCTION__);
85 mout << "illegal sliding mode single-horz, single-vert" << mout.endl;
86 fill = & SlidingWindow<C,R>::fillBoth;
87 }

223 void slideHorz(){

301 inline // FINAL
302 bool moveDown(){
303
304 ++this->location.y;
305 if (this->location.y <= this->coordinateHandler.getYRange().max){ // < this->srcHeight){
306 locationLead.y = this->location.y + this->jRange.max;
307 locationTrail.y = this->location.y + this->jRange.min - 1;
308 (this->*updateVert)();
309 return true;
310 }
311 else { // step back
312 --this->location.y;
313 return false;
314 }
315 }
316
318
323 inline // FINAL
324 bool moveUp(){
325
326 --this->location.y;
327 if (this->location.y >= 0){
328 locationLead.y = this->location.y + this->jRange.min;
329 locationTrail.y = this->location.y + this->jRange.max + 1;
330 (this->*updateVert)();
331 return true;
332 }
333 else { // step back
334 ++this->location.y;
335 return false;
336 }
337 }
338

345 inline // FINAL
346 bool moveRight(){
347
348 ++this->location.x;
349 if (this->location.x <= this->coordinateHandler.getXRange().max){ // this->srcWidth){
350 locationLead.x = this->location.x + this->iRange.max;
351 locationTrail.x = this->location.x + this->iRange.min-1;
352 (this->*updateHorz)();
