213 void BinaryPixelVectorOp<F>::traverseChannels(const ImageTray<const Channel> & src1, const ImageTray<const Channel> & src2,
214  ImageTray<Channel> & dst) const {
215 
216 
217  Logger mout(getImgLog(), __FUNCTION__, __FILE__); //REPL getImgLog(), name+"(PixelVectorOp)", __FUNCTION__);
218 
219  if (src1.empty())
220  mout.error() << "src1 empty" << mout.endl;
221 
222  if (src2.empty())
223  mout.error() << "src2 empty" << mout.endl;
224 
225  if (dst.empty())
226  mout.error() << "dst empty" << mout.endl;
227 
228  Channel & dstChannel = dst.get();
229 
230  if (src1.getGeometry() != src2.getGeometry()){
231  mout.warn() << "src1" << src1.getGeometry() << mout.endl;
232  mout.warn() << "src2" << src2.getGeometry() << mout.endl;
233  mout.note() << "dstChannel" << dstChannel << mout.endl;
234  mout.error() << "src1 and src2 geometry error" << mout.endl;
235  return;
236  }
237 
238  const int width = src1.getGeometry().getWidth(); //std::min(src1.getWidth(),src2.getWidth());
239  const int height = src1.getGeometry().getHeight(); //std::min(src1.getHeight(),src2.getHeight());
240  const size_t channels = src1.size(); // std::min(src1.getChannelCount(),src2.getChannelCount());
241 
242  double x=0.0, sum=0.0;
243 
244  mout.debug3() << *this << mout.endl;
245  //mout.debug3() << "src1: " << src1 << mout.endl;
246  //mout.debug3() << "src2: " << src2 << mout.endl;
247  //mout.debug3() << "dst: " << dst << mout.endl;
248 
249  FunctorBank & functorBank = getFunctorBank();
250 
251  std::string functorName;
252  std::string functorParams;
253 
254  drain::StringTools::split2(functorDef, functorName, functorParams, "/");
255 
256 
257  if (!functorBank.has(functorName)){
258  //functorBank.help(std::cerr);
259  mout.note() << functorBank << mout.endl;
260  mout.error() << "functor '" << functorName << "' not found: " << functorDef << mout.endl;
261  return;
262  }
263 
264  UnaryFunctor & scalingFunctor = functorBank.clone(functorName); // consider get? Or static from clone()
265 
266  //mout.debug() << scalingFunctor.getName() << ':' << scalingFunctor.getDescription() << mout.endl;
267  //scalingFunctor.setScale(dstChannel.scaling.getMax<double>(), 0.0);
268  const std::type_info & t = dstChannel.getType();
269  if (Type::call<drain::typeIsSmallInt>(t)){
270  const double m = Type::call<typeMax, double>(t);
271  scalingFunctor.setScale(m, 0.0); // TODO consider invScale, invOffset?
272  mout.info() << "small int type, scaling with its maximum (" << m << ')' << mout;
273  }
274 
275  //scalingFunctor.setParameters(functorParams);
276  try {
277  scalingFunctor.setParameters(functorParams, '=', '/');
278  } catch (const std::exception & e) {
279  mout.info(scalingFunctor);
280  mout.warn(scalingFunctor.getParameters());
281  mout.error(e.what());
282  }
283 
284  mout.debug2() << scalingFunctor.getName() << ':' << scalingFunctor << mout;
285 
286  const double coeff = (rescale>0.0) ? 1.0/rescale : 1.0/static_cast<double>(channels);
287  const bool USE_POW = (POW != 1.0);
288  const bool USE_INVPOW = (INVPOW != 1.0);
289  mout.debug3() << "coeff " << coeff << mout;
290  const BinaryFunctor & f = binaryFunctor;
291  size_t a;
292  for (int j = 0; j < height; j++) {
293  for (int i = 0; i < width; i++) {
294 
295  sum = 0.0;
296  a = dstChannel.address(i,j);
297  for (size_t k = 0; k < channels; k++){
298  // getValue( src1.get(k).get<double>(a) , src2.get(k).get<double>(a) );
299  x = f(src1.get(k).get<double>(a) , src2.get(k).get<double>(a)) ;
300  if (USE_POW)
301  sum += pow(x, POW);
302  else
303  sum += x;
304  }
305  if (USE_INVPOW)
306  dstChannel.put(a, scalingFunctor(pow(coeff*sum, INVPOW)));
307  else
308  dstChannel.put(a, scalingFunctor(coeff*sum));
309  }
310  }
