356 bool DataSelector::getPaths(const Hi5Tree & src, T & pathContainer) const {
357 
358  drain::Logger mout(__FUNCTION__, __FILE__);
359 
360  // Current search point
361  const Hi5Tree & s = src;
362 
364  std::set<ODIMPathElem::index_t> dataSetIndices;
365 
366  for (Hi5Tree::const_iterator it = s.begin(); it != s.end(); ++it) {
367 
368  const ODIMPathElem & currentElem = it->first;
369  mout.debug3() << "currentElem='" << currentElem << "'" << mout.endl;
370 
371  // can be moved inside DATASET scope if no other groups included in path (in future?)
372  const drain::image::Image & data = it->second.data.dataSet; // for ODIM
373  const drain::FlexVariableMap & props = data.getProperties();
374 
375  ODIMPath path;
376  path << currentElem;
377 
378 
379  // Check ELANGLE (in datasets) or quantity (in data/quality)
380  if (currentElem.is(ODIMPathElem::DATASET)){
381 
382  if (dataSetIndices.size() >= count){
383  mout.debug2() << "dataset count " << dataSetIndices.size() << ">=" << count << ") already completed for " << currentElem << mout.endl;
384  continue;
385  }
386 
387 
388  if (props.hasKey("where:elangle")){
389  double e = props["where:elangle"];
390  if (!elangle.contains(e)){
391  mout.debug("elangle ",e," outside range ",elangle);
392  continue;
393  }
394  }
395 
396  if (getInnerPaths(src, pathContainer, path)){
397  // Add this data set path
398  if (pathMatcher.match(path)){
399  mout.debug2() << " path matches (subtree OK) " << path << mout;
400  addPath3(pathContainer, props, path);
401  }
402  dataSetIndices.insert(currentElem.index);
403  }
404 
405  //continue;
406  }
407  //else if (currentElem.belongsTo(ODIMPathElem::DATA | ODIMPathElem::DATA)){
408  else if (currentElem.belongsTo(ODIMPathElem::DATA | ODIMPathElem::QUALITY)){ // 2021/04
409 
410  if (currentElem.is(ODIMPathElem::DATA)){
411  mout.warn() << " DATA_GROUP '" << currentElem << "' directly under root" << mout;
412  }
413 
414  if (getInnerPaths(src, pathContainer, ODIMPath())){
415  // added qualityX ...
416  }
417 
418  }
419  else if (currentElem.belongsTo(ODIMPathElem::ATTRIBUTE_GROUPS)){
420  // Consider policy..
421  //mout.warn() << " skipping ATTRIBUTE_GROUP: /" << currentElem << mout;
422  if (pathMatcher.match(path)){
423  addPath3(pathContainer, props, path);
424  }
425  // addPath3(pathContainer, v, path); // ?
426  }
427  else {
428  mout.warn() << " skipping odd group: /" << currentElem << mout;
429  }
430 
431  }
432 
433  return true; //...
434 }

437 template <class T>
438 bool DataSelector::getInnerPaths(const Hi5Tree & src, T & pathContainer, const ODIMPath & path) const {
439 
440  drain::Logger mout(__FUNCTION__, __FILE__);
441 
442  const bool quantityRequired = quantityRegExp.isSet() || qualityRegExp.isSet();
443 
444  bool groupsOK = false || !quantityRequired; // :-)
445 
446  const Hi5Tree & s = src(path);
447 
448  for (Hi5Tree::const_iterator it = s.begin(); it != s.end(); ++it) {
449 
450  bool quantityFound = false || !quantityRequired; // :-)
451 
452  //bool quantityOk = !quantityRequired; // Always ok if not required...
453  const ODIMPathElem & currentElem = it->first;
454  mout.debug3() << "currentElem='" << currentElem << "'" << mout.endl;
455  const drain::image::Image & data = it->second.data.dataSet; // for ODIM
456  const drain::FlexVariableMap & props = data.getProperties();
457 
458  ODIMPath p(path);
459  p << currentElem; // also for attrib groups
460 
461  if (currentElem.belongsTo(ODIMPathElem::DATA | ODIMPathElem::QUALITY)){
462 
463  if (quantityRequired){
464 
465 
466 
467  if (props.hasKey("what:quantity")){
468 
469  const std::string quantity = props["what:quantity"].toStr();
470 
471  if (qualityRegExp.isSet()){ // NOTE: at least, do not test quantity after this!
472  //quantityOk = false;
473  if (currentElem.is(ODIMPathElem::QUALITY) && qualityRegExp.test(quantity)){
474  mout.note() << "QUALITY quantity matches: [" << quantity << "]" << mout.endl;
475  //quantityOk = true;
476  quantityFound = true;
477  }
478  }
479  else if (quantityRegExp.test(quantity)){
480  mout.debug2() << "quantity matches: [" << quantity << "]: " << path << '|' << currentElem << mout.endl;
481  //quantityOk = true;
482  quantityFound = true;
483  }
484  else {
485  mout.debug3() << "unmatching DATA quantity [" << quantity << "], skipping" << mout.endl;
486  //NO continue; have to descend for quality (QIND) below!
487  //quantityOk = false;
488  // continue;
489  // recursion continues below
490  }
491 
492  if (quantityFound && (dualPRF != 0)){
493  double lowPRF = props.get("how:lowprf", 0.0);
494  double hightPRF = props.get("how:highprf", lowPRF);
495  bool IS_DUAL_PRF = (lowPRF != hightPRF);
496  if (dualPRF > 0){
497  if (!IS_DUAL_PRF){
498  mout.experimental() << "dualPRF required, rejecting [" << quantity << "] at " << p << mout;
499  quantityFound = false;
500  }
501  }
502  else {
503  if (IS_DUAL_PRF){
504  mout.experimental() << "dualPRF rejected: [" << quantity << "] at " << p << mout;
505  quantityFound = false;
506  }
507  }
508  }
509 
510  }
511  else {
512 
513  if (currentElem.is(ODIMPathElem::DATA))
514  mout.warn();
515  else
516  mout.info();
517  mout << "quantity missing in (image) metadata of " << path << '/' << currentElem << mout.endl;
518 
519  // continue?
520  }
521 
522  }
523 
524 
525  if (quantityFound && pathMatcher.match(p)){
526  addPath3(pathContainer, props, p);
527  }
528 
529  // Recursion, possibly finding quality quantity
530  quantityFound |= getInnerPaths(src, pathContainer, p);
531 
532  }
533  else if (currentElem.belongsTo(ODIMPathElem::ATTRIBUTE_GROUPS | ODIMPathElem::ARRAY)){
534  // mout.warn() << " ATTRIBUTE_GROUP: " << path << " / " << currentElem << mout;
535  if (pathMatcher.match(p)){
536  mout.debug3() << " adding" << path << " / " << currentElem << mout;
537  addPath3(pathContainer, props, p);
538  }
539  }
540  else if (currentElem.is(ODIMPathElem::DATASET)){
541  mout.warn() << " NESTING DATASET_GROUP: " << path << " / " << currentElem << mout;
542  // addPath3(pathContainer, v, path); // ?
543  }
544  else {
545  mout.debug() << " skipping special: " << path << " / " << currentElem << mout;
546  }
547  /*
548  else if (currentElem.is(ODIMPathElem::ARRAY)){
549  if (pathMatcher.match(p)){
550  mout.debug3() << " adding" << path << " / " << currentElem << mout;
551  addPath3(pathContainer, props, p);
552  }
553  }
554 
555  */
556  groupsOK |= quantityFound;
557  }
558 
559  return groupsOK;
560 }
