1042 bool hasPath(typename path_t::const_iterator it, typename path_t::const_iterator eit) const {
1043
1044 if (it == eit) // empty path
1045 return true;
1046
1047 const typename path_t::elem_t elem = *it;
1048
1049 if (elem.empty())
1050 return hasPath(++it, eit);
1051 else if (!hasChild(elem))
1052 return false;
1053
1054 return this->operator [](elem).hasPath(++it, eit);
1055
1056 }
