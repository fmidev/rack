963  template <class T>
964  void assignContainer(const T & v, bool append=false) {
965 
966  if (!append){
967  if (isStlString())
968  clear();
969  else
970  requestSize(0);
971  }
972 
973  suggestType(typeid(typename T::value_type));
974  // Note: sequence of strings, will request string type...
975 
976  if (isString()){
977  std::stringstream sstr;
978  if (append){
979  drain::Sprinter::toStream(sstr, *this, drain::Sprinter::plainLayout);
980  //toStream(sstr); // outputsepp?
981  if (inputSeparator)
982  sstr << inputSeparator;
983  }
984  StringTools::join(v, sstr, inputSeparator); // implement!
985  assignToString(sstr.str());
986  }
987  else if (caster.typeIsSet()){
988 
989  if (append){
990  for (typename T::const_iterator it = v.begin(); it != v.end(); ++it){
991  appendToElementArray(*it);
992  }
993  }
994  else {
995  requestSize(v.size());
996  typename T::const_iterator it = v.begin();
997  typename T::const_iterator itLast = it;
998  for (size_t i = 0; i<getElementCount(); ++i){
999  if (it == v.end()){
1000  if (fillArray)
1001  it = itLast; //v.begin(); WHAT if empty?
1002  else
1003  return;
1004  }
1005  else {
1006  itLast = it;
1007  }
1008  caster.put(getPtr(i), *it);
1009  ++it;
1010  }
1011  }
1012  }
1013  else {
1014  throw std::runtime_error(std::string(__FILE__) + ": type unset, cannot assign");
1015  }
