546 std::string Caster::get<std::string>(const void *p) const {
547 
548  // void is handled by toOStr, but this is faster.
549  if (getType() == typeid(void)){

551  return empty;
552  }
553  else if (getType() == typeid(char)){
554  // std::cout << "note: experimental char to str\n";

557  else if (getType() == typeid(std::string)){
558  return *(const std::string *)p;
559  }
560  else {
561  std::stringstream sstr;
562  toOStream(sstr, p);
563  // sstr << (this->getDouble)(p);
564  return sstr.str();
565  }
